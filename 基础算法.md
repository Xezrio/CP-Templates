### **快速排序**

```c++
void quicksort(int q[], int l, int r){
    if(l >= r) return;
    int x = q[l + r >> 1], i = l - 1, j = r + 1;
    while(i < j){
        do i++; while(q[i] < x);
        do j--; while(q[j] > x);
        if(i < j) swap(q[i], q[j]);
    }
    quicksort(q, l, j);
    quicksort(q, j + 1, r);
}
```



### **归并排序**

```c++
int tmp[N], a[N];

void merge_sort(int q[], int l, int r){
    if(l >= r) return;
    
    int mid = l + r >> 1;
    merge_sort(q, l, mid);
    merge_sort(q, mid + 1, r);

    int k = 1, i = l, j = mid + 1;
    while(i <= mid && j <= r){
        if(q[i] <= q[j]) tmp[k++] = q[i++];
        else tmp[k++] = q[j++];
    }
    while(i <= mid) tmp[k++] = q[i++];
    while(j <= r) tmp[k++] = q[j++];

    for(int i = l, j = 1;i <= r;++i, ++j) q[i] = tmp[j];
}
```

浮点二分（数的三次方根）

```cpp
double float_binary_search(double n, double l, double r){
    constexpr double eps = 1e-8;
    while(r - l > eps){
        double mid = (l + r) / 2;
        if(pow(mid, 3) >= n) r = mid;
        else l = mid;
    }
    return l;
}
```

```cpp
bool cmp(vector<int> &a, vector<int> &b){
    if(a.size() != b.size()) return a.size() > b.size();
    else{
        for(int i = a.size() - 1;i >= 0;--i){
            if(a[i] != b[i]) return a[i] > b[i];
        }
    }
    return true;
}

vector<int> sub(vector<int> &a, vector<int> &b){
    vector<int> C;
    int t = 0;
    for(int i = 0;i < a.size();++i){
        t = a[i] - t;
        if(i < b.size()) t -= b[i];
        C.push_back((t + 10) % 10);
        if(t < 0) t = 1;
        else t = 0;
    }
    while (C.size() > 1 && C.back() == 0) C.pop_back();
    return C;
}

void solve(){

}
```

```cpp
void insert(int x1, int y1, int x2, int y2, int c){
```

### 双指针

最长连续不重复子序列

```cpp
int a[N];
int cnt[N];

void solve(){
    int n;
    cin >> n;
    for(int i = 1;i <= n;++i) cin >> a[i];
    int res = 0;
    for(int i = 1, j = 1;i <= n;++i){
        cnt[a[i]]++;
        while(cnt[a[i]] > 1){
            cnt[a[j]]--;
            j++;
        }
        res = max(res, i - j + 1);
    }
    cout << res << '\n';
}
```
