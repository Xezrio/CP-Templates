### 表达式求值

```cpp
stack<int> num;
stack<char> op;

void calc(){
    int res = 0;
    int b = num.top(); num.pop();
    int a = num.top(); num.pop();
    char x = op.top(); op.pop();
    if(x == '+') res = a + b;
    if(x == '-') res = a - b;
    if(x == '*') res = a * b;
    if(x == '/') res = a / b;
    num.push(res);
}

void solve(){
    string s;
    cin >> s;
    map<char, int> pri;
    pri['+'] = pri['-'] = 1;
    pri['*'] = pri['/'] = 2;

    for(int i = 0;i < s.size();++i){
        if(isdigit(s[i])){
            int j = i;
            while(j < s.size() && isdigit(s[j])) j++;
            num.push(stoi(s.substr(i, j - i)));
            i = j - 1;
        }
        else if(s[i] == '(') op.push(s[i]);
        else if(s[i] == ')'){
            while(op.top() != '(') calc();
            op.pop();
        }
        else{
            while(op.size() && pri[s[i]] <= pri[op.top()]) calc();
            op.push(s[i]);
        }
    }
    while(op.size()) calc();
    int res = num.top();
    cout << res << '\n';
}
```



### 滑动窗口

```cpp
// q中存储窗口有效元素的下标
int a[N], q[N], hh, tt = -1; 

void solve(){
    int n, k;
    cin >> n >> k;
    for(int i = 0;i < n;++i) cin >> a[i];
    for(int i = 0;i < n;++i){
        if(i - k + 1 > q[hh]) hh++; // 弹出队头
        while(hh <= tt && a[q[tt]] >= a[i]) tt--; // hh <= tt防止tt滑出窗口
        q[++tt] = i;
        if(i >= k - 1) cout << a[q[hh]] << ' '; // 0 ~ k-1 有k项
    }
    cout << '\n';

    hh = 0, tt = - 1;   // 别忘了初始化
    for(int i = 0;i < n;++i) cin >> a[i];
    for(int i = 0;i < n;++i){
        if(i - k + 1 > q[hh]) hh++;
        while(hh <= tt && a[q[tt]] <= a[i]) tt--;
        q[++tt] = i;
        if(i >= k - 1) cout << a[q[hh]] << ' ';
    }
    cout << '\n';
}
```

```cpp
char p[M], s[N];
int ne[N];

void kmp(){
    int m, n;
    cin >> m >> p + 1 >> n >> s + 1; // 下标从1开始

    for(int i = 2, j = 0;i <= m;++i){ // ne[1] = 0, 可以直接从2开始
        while(j && p[i] != p[j + 1]) j = ne[j];
        if(p[i] == p[j + 1]) j++;
        ne[i] = j;
    }

    for(int i = 1, j = 0;i <= n;++i){
        while(j && s[i] != p[j + 1]) j = ne[j]; // j没到尽头0而且不匹配
        if(s[i] == p[j + 1]) j++;
        if(j == m){
            cout << i - m + 1 - 1 << ' '; // 下标从0开始要-1
            j = ne[j]; // 回退j准备下一轮匹配
        }
    }
}
```

```cpp
constexpr int N = 1e5+10, M = 31*N;

int a[N];
int son[M][2], idx;

void insert(int x){
    int p = 0;
    for(int i = 30;i >= 0;--i){
        int u = x >> i & 1;
        if(!son[p][u]) son[p][u] = ++idx;
        p = son[p][u];
    }
}

int query(int x){
    int res = 0;
    int p = 0;
    for(int i = 30;i >= 0;--i){
        int u = x >> i & 1;
        if(son[p][!u]){
            p = son[p][!u];
            res = res * 2 + 1;
        }
        else{
            p = son[p][u];
            res = res * 2 + 0;
        }
    }
    return res;
}

M: 节点的总个数

p: 当前父节点
u: 当前要检查的子节点（如字符串是当前字母，二进制是0或1）
根节点: idx = 0
```

食物链

```cpp
int p[N], d[N];

// 路径压缩
int find(int x){
    if(p[x] != x){
        int t = find(p[x]);
        d[x] += d[p[x]]; 
        p[x] = t;
    }
    return p[x];
}

void solve(){
    int n, m;
    cin >> n >> m;

    // 初始化并查集
    for(int i = 1;i <= n;++i) p[i] = i;

    int ans = 0;
    while(m--){
        int t, x, y;
        cin >> t >> x >> y;
        if(x > n || y > n){ans++; continue;}

        int px = find(x), py = find(y);
        if(t == 1){
            if(px == py && (d[x] - d[y]) % 3) ans++;
            else if(px != py){
                p[px] = py;
                d[px] = d[y] - d[x];
            }
        }
        else{
            if(px == py && (d[x] - d[y] - 1) % 3) ans++;
            else if(px != py){
                p[px] = py;
                d[px] = d[y] + 1 - d[x];
            }
        }
    }
    cout << ans << '\n';
}
```



### 字符串哈希

```cpp
using ull = unsigned long long;

constexpr int N = 1e5+10; // 字符串长度
constexpr int P = 131; // 记得131对应2^64，开ull

ull h[N], p[N];
string str;

ull gethash(int l, int r){ // 哈希值开ull
    return h[r] - h[l-1] * p[r-l+1];
}

void string_hash(){
    int n, m;
    cin >> n >> m >> str;
    str = " " + str; // str + 1
    p[0] = 1;   // 注意初始化 P ^ 0 = 1
    for(int i = 1;i <= n;++i){
        p[i] = p[i-1] * P;
        h[i] = h[i-1] * P + str[i];
    }
    while(m--){
        int l1, r1, l2, r2;
        cin >> l1 >> r1 >> l2 >> r2;
        if(gethash(l1, r1) == gethash(l2, r2)) cout << "Yes\n";
        else cout << "No\n";
    }
}
```
