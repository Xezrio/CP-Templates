```cpp
int cnt;
int primes[N];
bool st[N];

void Eula(int x){
    for(int i = 2;i <= x;++i){
        if(!st[i]) primes[++cnt] = i;
        for(int j = 1;primes[j] <= x / i;++j){
```

注释

```cpp
void Eula(int x){
    for(int i = 2;i <= x;++i){ // 每轮用质因子乘i筛合数
        if(!st[i]) primes[++cnt] = i;
        for(int j = 1;primes[j] <= x / i;++j){  // primes[j] * i <= x 只筛掉<= x的合数
            st[primes[j] * i] = true; // 用最小质因子筛合数
            if(i % primes[j] == 0) break; // 每个合数只用它的最小质因数筛
        }
    }
}
// 如果i % primes[j] != 0，说明此时的质数primes[j]不是i的质因子，质数从小到大遍历，
// 说明这个质数小于i的最小质因子，所以primes[j] * i的最小质因子是primes[j]
// 如果i % primes[j] == 0，说明此时质数primes[j]是i的最小质因子，
// 那么primes[j] * i的最小质因子也是primes[j]，primes[j+1] * i的最小质因子是primes[j]而不是primes[j+1]，为保证每个合数都只被它的最小质因数筛，所以break;
```

### 欧拉函数

> $1 \sim N$ 中与 $N$ 互质的数的个数被称为欧拉函数，记为 $ϕ(N)$。 
> 若在算数基本定理中，$N = p_1^{a_1}p_2^{a_2}…p_m^{a_m}$，则： 
> $ϕ(N)$ = $N \times \frac{p_1-1}{p_1} \times \frac{p_2-1}{p_2} \times … \times \frac{p_m-1}{p_m}$

```cpp
int phi(int x){
    int res = x; // x * II
    for(int i = 2;i <= x / i;++i){
        if(x % i == 0){
            res = res / i * (i - 1); // 先除再乘（保证res不小于i）
            while(x % i == 0) x /= i;
        }
    }
    if(x > 1) res = res / x * (x - 1); // 分解质因数可能有一项大于x / i
    return res;
}
```

从$1$到$p_s^{a_s}$中共有$p_s^{a_s}$个数字

其中与$p_s^{a_s}$不互质的有$p_s, 2p_s, …, {p_s}^{{a_s}- 1} * p_s $ ,共${p_s}^{ a_s - 1} $ 项

所以 $\varphi(p_s^{a_s})$ = $p_s^{a_s}$ - ${p_s}^{{a_s}- 1} = p_s^{a_s} * (1 - \frac {1}{p_s})$ 



### 筛法求欧拉函数

```cpp
constexpr int N = 1e6+10;

int primes[N], euler[N], cnt;
bool st[N];

void get_eulers(int x){
    euler[1] = 1;
    for(int i = 2;i <= x;++i){
        if(!st[i]){
            primes[++cnt] = i;
            euler[i] = i - 1;
        }
        for(int j = 1;primes[j] <= x / i;++j){
            int t = primes[j] * i;
            st[t] = true;
            if(i % primes[j] == 0){
                euler[t] = primes[j] * euler[i];
                break;
            }
            else euler[t] = (primes[j] - 1) * euler[i];
        }
    }
}

void solve(){
    int x;
    cin >> x;

    get_eulers(x);

    ll ans = 0;
    for(int i = 1;i <= x;++i) ans += euler[i];
    cout << ans << '\n';
}
```

```cpp
    if(!b){
        x = 1, y = 0;

    return d;
```

### 扩展中国剩余定理

给定 $2n$ 个整数 $a_1,a_2,…,a_n$ 和 $m_1,m_2,…,m_n$，求一个最小的非负整数 $x$，满足 $ \forall i \in [1,n],x \equiv m_i(mod\ a_i)$ 。

输出最小非负整数 $x$，如果 $x$ 不存在，则输出 $-1$。

```cpp
using ll = long long;

ll exgcd(ll a, ll b, ll &x, ll &y){
    if(!b){
        x = 1, y = 0;
        return a;
    }
    ll d = exgcd(b, a % b, y, x);
    y -= a / b * x;
    return d;
}

inline ll qmul(ll a, ll b, ll m){ // 快速乘取余
    ll res = 0;
    a = (a % m + m) % m, b = (b % m + m) % m;
    while(b){
        if(b & 1) res = (res + a) % m;
        b >>= 1;
        a = (a + a) % m;
    }
    return res;
}

ll EXCRT(){
    int n;
    cin >> n;

    ll m1, a1;
    cin >> m1 >> a1;

    int tt = n - 1;
    while(tt--){
        ll m2, a2;
        cin >> m2 >> a2;
        ll k1, k2;
        ll d = exgcd(m1, m2, k1, k2);

        if((a2 - a1) % d) return -1; // 无解

        k1 = qmul(k1, (a2 - a1) / d, m2 / d);
        a1 = k1 * m1 + a1;
        m1 = m1 / d * m2; // lcm(m1, m2)
    }

    return (a1 % m1 + m1) % m1;
}
```



### 高斯消元解线性方程组

输入一个包含 $n$ 个方程 $n$ 个未知数的线性方程组。 方程组中的系数为实数。 求解这个方程组。 下图为一个包含 $m$ 个方程 $n$ 个未知数的线性方程组示例： 

![9a504fc2d5628535be9dcb5f90ef76c6a7ef634a.gif](https://cdn.acwing.com/media/article/image/2019/06/27/19_b30080c698-9a504fc2d5628535be9dcb5f90ef76c6a7ef634a.gif)

```cpp
constexpr int N = 110, INF = 0x3f3f3f3f;
constexpr double eps = 1e-8;

int n;
double a[N][N];

int gauss(){
    int c, r;
    for(c = 1, r = 1;c <= n;++c){
        int t = r;
        for(int i = r;i <= n;++i)
            if(fabs(a[i][c]) > fabs(a[t][c])) t = i;

        if(fabs(a[t][c]) < eps) continue;

        for(int i = c;i <= n + 1;++i) swap(a[t][i], a[r][i]);
        for(int i = n + 1;i >= c;--i) a[r][i] /= a[r][c];
        for(int i = r + 1;i <= n;++i){
            if(fabs(a[i][c]) > eps)
                for(int j = n + 1;j >= c;--j) a[i][j] -= a[i][c] * a[r][j];
        }
        r++;
    }
    if(r <= n){
        for(int i = r;i <= n;++i){
            if(fabs(a[i][n+1]) > eps) return 0;
        }
        return INF;
    }
    for(int i = n;i >= 1;--i){
        for(int j = i + 1;j <= n;++j)
            a[i][n+1] -= a[i][j] * a[j][n+1];
    }
    return 1;
}

void solve(){
    cin >> n;
    for(int i = 1;i <= n;++i)
        for(int j = 1;j <= n + 1;++j)
            cin >> a[i][j];

    int t = gauss();
    if(t == 0) cout << "No solution" << '\n';
    else if(t == INF) cout << "Infinite group solutions" << '\n';
    else{
        for(int i = 1;i <= n;++i)
            cout << fixed << setprecision(2) << a[i][n+1] << '\n';
    }
}
```



### 高斯消元解异或线性方程组

输入一个包含 $n$ 个方程 $n$ 个未知数的异或线性方程组。 方程组中的系数和常数为 $0$ 或 $1$，每个未知数的取值也为 $0$ 或 $1$。 求解这个方程组。 异或线性方程组示例如下： 

``` M[1][1]x[1] ^ M[1][2]x[2] ^ … ^ M[1][n]x[n] = B[1] ``` 

 ```M[2][1]x[1] ^ M[2][2]x[2] ^ … ^ M[2][n]x[n] = B[2]```

 ```… ```

 ```M[n][1]x[1] ^ M[n][2]x[2] ^ … ^ M[n][n]x[n] = B[n] ```

 其中 `^` 表示异或($XOR$)，$M[i][j]$ 表示第 $i$ 个式子中 $x[j]$ 的系数，$B[i]$ 是第 $i$ 个方程右端的常数，取值均为 $0$ 或 $1$。

```cpp
constexpr int N = 110, INF = 0x3f3f3f3f;

int n;
int a[N][N];

void out(){
    for(int i = 1;i <= n;++i){
        for(int j = 1;j <= n + 1;++j)
            cout << a[i][j] << ' ';
        cout << '\n';
    }
    cout << '\n';
}

int gauss(){
    int c, r;
    for(c = 1, r = 1;c <= n;++c){
        int t = r;
        for(int i = r;i <= n;++i)
            if(a[i][c]){
                t = i;
                break;
            }

        if(!a[t][c]) continue;

        for(int i = c;i <= n + 1;++i) swap(a[t][i], a[r][i]);
        for(int i = r + 1;i <= n;++i){
            if(a[i][c])
                for(int j = n + 1;j >= c;--j) a[i][j] ^= a[r][j];
        }
        r++;
    }
    if(r <= n){
        for(int i = r;i <= n;++i){
            if(a[i][n+1]) return 0;
        }
        return INF;
    }
    for(int i = n;i >= 1;--i){
        for(int j = i + 1;j <= n;++j)
            if(a[i][j]) a[i][n+1] ^= a[j][n+1];
    }
    return 1;
}

void solve(){
    cin >> n;
    for(int i = 1;i <= n;++i)
        for(int j = 1;j <= n + 1;++j)
            cin >> a[i][j];

    int t = gauss();
    if(t == 0) cout << "No solution" << '\n';
    else if(t == INF) cout << "Multiple sets of solutions" << '\n';
    else{
        for(int i = 1;i <= n;++i)
            cout << a[i][n+1] << '\n';
    }
}
```

```cpp
ll p;

int qpow(int a, int b){

        b >>= 1;

int C(int a, int b){
    int res = 1;
    for(int i = 1, j = a;i <= b;++i,--j){
```

### 求组合数4

输入 $a, b$，求 $C_a^b$ 的值。 注意结果可能很大，需要使用高精度计算。

```cpp
constexpr int N = 5050;

int sum[N];
int primes[N], cnt;
bool st[N];

void get_primes(int x){
    for(int i = 2;i <= x;++i){
        if(!st[i]) primes[++cnt] = i;
        for(int j = 1;primes[j] <= x / i;++j){
            st[primes[j] * i] = true;
            if(i % primes[j] == 0) break;
        }
    }
}

int getSum(int n, int p){
    int res = 0;
    while(n){
        res += n / p;
        n /= p;
    }
    return res;
}

vector<int> mul(vector<int> a, int b){
    vector<int> c;
    int t = 0;
    for(int i = 0;i < a.size() || t;++i){
        if(i < a.size()) t += a[i] * b;
        c.push_back(t % 10);
        t /= 10;
    }
    return c;
}

void solve(){
    int a, b;
    cin >> a >> b;
    get_primes(a);
    for(int i = 1;i <= cnt;++i){
        int p = primes[i];
        sum[i] = getSum(a, p) - getSum(b, p) - getSum(a - b, p);
    }
    vector<int> res;
    res.emplace_back(1);
    for(int i = 1;i <= cnt;++i){
        for(int j = 1;j <= sum[i];++j)
            res = mul(res, primes[i]);
    }
    for(int i = res.size() - 1;i >= 0;--i) cout << res[i]; cout << '\n';
}
```



### 能被整除的数、二进制枚举

给定一个整数 $n$ 和 $m$ 个不同的质数 $p_1, p_2, …, p_m$。 请你求出 $1 \sim n$ 中能被 $p_1, p_2, …, p_m$ 中的至少一个数整除的整数有多少个。

```cpp
constexpr int N = 20;

int p[N];

void solve(){
    int n, m;
    cin >> n >> m;
    for(int i = 0;i < m;++i) cin >> p[i];
    // 此处省去i = 0即全不选的情况
    ll res = 0;
    for(int i = 1;i < 1 << m;++i){
        int t = 1, cnt = 0;
        for(int j = 0;j < m;++j){
            if(i >> j & 1){
                if((ll)t * p[j] > n){
                    t = -1;
                    break;
                }
                t *= p[j];
                cnt++;
            }
        }
        if(t != -1){
            if(cnt & 1) res += n / t;
            else res -= n / t;
        }
    }
    cout << res << '\n';
}
```

### 台阶-Nim游戏

现在，有一个 $n$ 级台阶的楼梯，每级台阶上都有若干个石子，其中第 $i$ 级台阶上有 $a_i$ 个石子($i \ge 1$)。 两位玩家轮流操作，每次操作可以从任意一级台阶上拿若干个石子放到下一级台阶中（不能不拿）。 已经拿到地面上的石子不能再拿，最后无法进行操作的人视为失败。 问如果两人都采用最优策略，先手是否必胜。

```cpp
void solve(){
    int n;
    cin >> n;
    int res = 0;
    for(int i = 1;i <= n;++i){
        int x;
        cin >> x;
        if(i & 1) res ^= x;
    }
    cout << (res ? "Yes" : "No") << '\n';
}
```



### 集合-Nim游戏

给定 $n$ 堆石子以及一个由 $k$ 个不同正整数构成的数字集合 $S$。 现在有两位玩家轮流操作，每次操作可以从任意一堆石子中拿取石子，每次拿取的石子数量必须包含于集合 $S$，最后无法进行操作的人视为失败。 问如果两人都采用最优策略，先手是否必胜。

```cpp
constexpr int N = 110, M = 10010;

int n, k;
int s[N], f[M];

int sg(int x){
    if(f[x] != -1) return f[x];
    unordered_set<int> S;
    for(int i = 1;i <= k;++i){
        if(x >= s[i]) S.insert(sg(x - s[i]));
    }
    for(int i = 0;;++i){
        if(!S.count(i)) return f[x] = i;
    }
}

void solve(){
    cin >> k;
    for(int i = 1;i <= k;++i) cin >> s[i];
    cin >> n;

    memset(f, -1, sizeof f);

    int res = 0;
    for(int i = 1;i <= n;++i){
        int x;
        cin >> x;
        res ^= sg(x);
    }
    cout << (res ? "Yes" : "No") << '\n';
}
```



### 拆分-Nim游戏

给定 $n$ 堆石子，两位玩家轮流操作，每次操作可以取走其中的一堆石子，然后放入两堆**规模更小**的石子（新堆规模可以为 $0$，且两个新堆的石子总数可以大于取走的那堆石子数），最后无法进行操作的人视为失败。 问如果两人都采用最优策略，先手是否必胜。

```cpp
constexpr int N = 110;

int f[N];

int sg(int x){
    if(f[x] != -1) return f[x];
    unordered_set<int> S;
    for(int i = 0;i < x;++i){
        for(int j = i;j < x;++j){
            S.insert(sg(i) ^ sg(j));
        }
    }
    for(int i = 0;;++i){
        if(!S.count(i)) return f[x] = i;
    }
}

void solve(){
    int n;
    cin >> n;

    memset(f, -1, sizeof f);

    int res = 0;
    while(n--){
        int x;
        cin >> x;
        res ^= sg(x);
    }
    cout << (res ? "Yes" : "No") << '\n';
}
```
