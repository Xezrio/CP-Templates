### BFS

走迷宫

```cpp
int n, m;
int g[N][N];
int d[N][N];
PII q[N * N];
PII Prev[N][N];

void bfs(){
    memset(d, -1, sizeof d);
    d[1][1] = 0;

    int hh = 1, tt = 1;
    q[1] = {1, 1};

    int dx[] = {0, 1, 0, -1}, dy[] = {1, 0, -1, 0};

    while(hh <= tt){
        auto t = q[hh++];
        for(int i = 0;i < 4;++i){
            int x = t.first + dx[i], y = t.second + dy[i];
            if(x >= 1 && x <= n && y >= 1 && y <= m && g[x][y] == 0 && d[x][y] == -1){
                d[x][y] = d[t.first][t.second] + 1;
                Prev[x][y] = t;
                q[++tt] = {x, y};
            }
        }
    }

    int x = n, y = m;
    while(x || y){
        g[x][y] = 4;
        auto t = Prev[x][y];
        x = t.first, y = t.second;
    }

    for(int i = 1;i <= n;++i){
        for(int j = 1;j <= m;++j){
            cout << g[i][j] << ' ';
        }
        cout << '\n';
    }
    cout << '\n';

    cout << d[n][m] << '\n';
}
```

```cpp
string start;

int bfs(){
    queue<string> q;
    unordered_map<string, int> d;

    q.push(start);
    d[start] = 0;

    int dx[] = {1, 0, -1, 0}, dy[] = {0, -1, 0, 1};

    string end = "12345678x";

    while(!q.empty()){
        auto t = q.front();
        q.pop();

        if(t == end) return d[t]; // t == end 返回

        int distance = d[t];
        int k = t.find('x'); // find返回x的下标
        int x = k / 3, y = k % 3; // 一维 -> 二维 x = k / 3, y = k % 3

        for(int i = 0;i < 4;++i){
            int a = x + dx[i], b = y + dy[i];
            if(a >= 0 && a < 3 && b >= 0 && b < 3){ // 别习惯性写了x, y，这里a, b才是偏移点
                swap(t[k], t[a * 3 + b]); // 二维 -> 一维 k = a * 3 + b

                if(!d[t]){
                    d[t] = distance + 1;
                    q.push(t);
                }

                swap(t[k], t[a * 3 + b]); // 记得恢复t，还有三个方向要搜
            }
        }
    }

    return -1;
}

void solve(){
    char c;
    for(int i = 0;i < 9;++i){
        cin >> c; // 一个一个字符，转成string
        start += c;
    }
    cout << bfs() << '\n';
}
```

### 有向图的拓扑序列

若一个由图中所有点构成的序列 $A$ 满足：对于图中的每条边 $(x,y)$，$x$ 在 $A$ 中都出现在 $y$ 之前，则称 $A$ 是该图的一个拓扑序列。

```Cpp
// d[]存储点的入度, q[]存储拓扑序列
int d[N], q[N]; 

bool topsort(){
    int hh = 1, tt = 0;
    for(int i = 1;i <= n;++i){
        if(!d[i]) q[++tt] = i;
    }

    while(hh <= tt){
        auto t = q[hh++]; // 队头不只有一次，所以记得写while(hh <= tt)
        for(int i = h[t];i != -1;i = ne[i]){ // 搜索队头h[t]的所有子节点
            int j = e[i]; // j才是需要的节点
            d[j]--;
            if(!d[j]) q[++tt] = j;
        }
    }
    return tt == n;
}
```

### 最短路

n为点数，m为边数

单源最短路

​		边权为正数

​				堆优化 $Dijkstra$		$O(mlog n)$

​		存在负权边

#### Dijkstra

##### 朴素

```cpp
constexpr int N = 520;

int g[N][N];
int n, m;
int dist[N];
bool st[N];

int dijkstra(){
    memset(dist, 0x3f, sizeof dist);
    dist[1] = 0; // 设置起点

    for(int i = 1;i <= n;++i){ // n个点迭代n次
        int t = -1;
        for(int j = 1;j <= n;++j){ // 找到未找到最短路的距离最近的点t
            if(!st[j] && (t == -1 || dist[j] < dist[t])){ // t未初始化或有距离1更近的点
                t = j;
            }
        }
        st[t] = true;
        for(int j = 1;j <= n;++j){ // 用t更新其他点的距离
            dist[j] = min(dist[j], dist[t] + g[t][j]);
        }
    }

    if(dist[n] == 0x3f3f3f3f) return -1;
    else return dist[n];
}
```

##### 堆优化

```cpp
constexpr int N = 2e5+10, M = N;

int n, m;
int h[N], e[M], w[M], ne[M], idx;
int dist[N];
bool st[N];

int dijkstra(){
    memset(dist, 0x3f, sizeof dist);
    dist[1] = 0; // 设置起点

    priority_queue<PII, vector<PII>, greater<PII>> heap; // 小根堆
    heap.push({0, 1}); // 别写反了
    while(heap.size()){
        auto t = heap.top(); // 存储距离1最近的点t
        heap.pop();
        int distance = t.first, ver = t.second;
        if(st[ver]) continue; // 去除距离不是最小的重边
        st[ver] = true;
        for(int i = h[ver];i != -1;i = ne[i]){
            int j = e[i];
            if(dist[j] > distance + w[i]){ // w[i] i到ver的权重
                dist[j] = distance + w[i];
                heap.push({dist[j], j}); // 把更新好距离的点存入堆，准备用作t更新
            }
        }
    }

    if(dist[n] == 0x3f3f3f3f) return -1;
    else return dist[n];
}
```

#### Bellman - Ford

##### 有边数限制的最短路

```cpp
constexpr int N = 510, M = 10010;

int n, m, k;
int dist[N], backup[N];

struct edge{
    int a, b, w;
}edges[M];

int bellman_ford(){
    memset(dist, 0x3f, sizeof dist);
    dist[1] = 0;

    for(int i = 1;i <= k;++i){ // 迭代k次，每次更新多考虑1条边
        memcpy(backup, dist, sizeof dist);
        for(int j = 1;j <= m;++j){
            int a = edges[j].a, b = edges[j].b, w = edges[j].w;
            dist[b] = min(dist[b], backup[a] + w);
            // 如果使用dist[a]可能会串联，就不能保证每次更新只多考虑一条边
        }
    }

    return dist[n];
}

void solve(){
    cin >> n >> m >> k;
    for(int i = 1;i <= m;++i){
        int a, b, w;
        cin >> a >> b >> w;
        edges[i] = {a, b, w};
    }
    int t = bellman_ford();
    if(t > 0x3f3f3f3f / 2) cout << "impossible" << '\n';
    else cout << t << '\n';
}
```



#### SPFA

##### 求最短路

```cpp
constexpr int N = 1e5+10, M = N;

int n, m;
int h[N], e[M], w[M], ne[M], idx;
int dist[N];
bool st[N]; // 当前队列有没有这个值

void add(int a, int b, int c){
    e[idx] = b, w[idx] = c, ne[idx] = h[a], h[a] = idx++;
}

int spfa(){
    memset(dist, 0x3f, sizeof dist);
    dist[1] = 0;

    queue<int> q;
    q.push(1);
    st[1] = true;

    while(q.size()){
        int t = q.front();
        q.pop();
        st[t] = false; // 出队记得标记
        for(int i = h[t];i != -1;i = ne[i]){
            int j = e[i];
            if(dist[j] > dist[t] + w[i]){
                dist[j] = dist[t] + w[i];
                if(!st[j]){ // 如果当前的队列里没有就入队
                    q.push(j);
                    st[j] = true;
                }
            }
        }
    }

    return dist[n];
}
```

##### 判断负环

```cpp
constexpr int N = 2010, M = 10010;

int n, m;
int h[N], e[M], w[M], ne[M], idx;
int dist[N], cnt[N];
bool st[N];

void add(int a, int b, int c){
    e[idx] = b, w[idx] = c, ne[idx] = h[a], h[a] = idx++;
}

int spfa(){
    queue<int> q; // 初始化dist为0，有负权边才更新
    for(int i = 1;i <= n;++i){
        q.push(i);
        st[i] = true;
    }

    while(q.size()){
        int t = q.front();
        q.pop();
        st[t] = false;
        for(int i = h[t];i != -1;i = ne[i]){
            int j = e[i];
            if(dist[j] > dist[t] + w[i]){
                dist[j] = dist[t] + w[i];

                cnt[j] = cnt[t] + 1; // cnt[j] 记录当前j点到虚拟源点最短路的边数
                if(cnt[j] >= n) return true; 
                // 每个点最多n - 1条边，超过n - 1条就是存在负环，有负环会转无穷多圈

                if(!st[j]){
                    q.push(j);
                    st[j] = true;
                }
            }
        }
    }
    return false;
}

memset(h, -1, sizeof h);
```

#### Floyd

```cpp
constexpr int N = 210, INF = 0x3f3f3f3f;

int n, m, q;
int d[N][N];

void floyd(){
    // k在最外层
    for(int k = 1;k <= n;++k){
        for(int i = 1;i <= n;++i){
            for(int j = 1;j <= n;++j){
                d[i][j] = min(d[i][j], d[i][k] + d[k][j]);
            }
        }
    }
}

void solve(){
    cin >> n >> m >> q;

    // 在输入之前初始化d[][]，否则输入值会被覆盖
    for(int i = 1;i <= n;++i) 
        for(int j = 1;j <= n;++j)
            if(i == j) d[i][j] = 0;
            else d[i][j] = INF;

    while(m--){
        int a, b, w;
        cin >> a >> b >> w;
        d[a][b] = min(d[a][b], w);
    }

    floyd();

    while(q--){
        int x, y;
        cin >> x >> y;
        if(d[x][y] > INF / 2) cout << "impossible" << '\n';
        else cout << d[x][y] << '\n';
    }
}
```

### 最小生成树

#### Prim

```cpp
constexpr int N = 520, INF = 0x3f3f3f3f;

int n, m;
int g[N][N];
int dist[N];
bool st[N];

int prim(){
    memset(dist, 0x3f, sizeof dist);
    int res = 0;
    for(int i = 1;i <= n;++i){
        int t = -1;
        for(int j = 1;j <= n;++j){
            if(!st[j] && (t == -1 || dist[j] < dist[t]))
                t = j;
        }

        if(i != 1 && dist[t] == INF) return INF;
        if(i != 1) res += dist[t]; // 写在更新dist前面防止更新负权自环
        st[t] = true;

        for(int j = 1;j <= n;++j) dist[j] = min(dist[j], g[t][j]);
    }
    return res;
}

void solve(){
    memset(g, 0x3f, sizeof g);
    cin >> n >> m;
    while(m--){
        int a, b, c;
        cin >> a >> b >> c;
        g[a][b] = g[b][a] = min(g[a][b], c); // 最小生成树无向图
    }
    int t = prim();
    if(t == INF) cout << "impossible" << '\n';
    else cout << t << '\n';
}
```

#### Kruskal

```cpp
constexpr int N = 1e5+10, M = 2e5+10, INF = 0x3f3f3f3f;

int n, m;
int p[N];

struct Edge{
    int a, b, w;
    bool operator<(const Edge &W)const{
        return this->w < W.w;
    }
}edges[M];

int find(int x){
    if(p[x] != x) p[x] = find(p[x]);
    return p[x];
}

int kruskal(){
    sort(edges+1, edges+m+1);
    for(int i = 1;i <= n;++i) p[i] = i; // 初始化并查集

    int res = 0, cnt = 0;
    for(int i = 1;i <= m;++i){
        int a = edges[i].a, b = edges[i].b, w = edges[i].w;
        int pa = find(a), pb = find(b);
        if(pa != pb){ // 可以排除自环
            p[pa] = pb;
            res += w;
            cnt++;
        }
    }
    if(cnt < n - 1) return INF; // n个点，连通至少需要n - 1条边
    else return res;
}

void solve(){
    cin >> n >> m;
    for(int i = 1;i <= m;++i){
        int a, b, c;
        cin >> a >> b >> c;
        edges[i] = {a, b, c};
    }
    int t = kruskal();
    if(t == INF) cout << "impossible" << '\n';
    else cout << t << '\n';
}
```



### 染色法判定二分图

```cpp
constexpr int N = 1e5+10, M = 2e5+10; // 无向图建边 M *= 2

int n, m;
int h[N], e[M], ne[M], idx;
int color[N];

void add(int a, int b){
    e[idx] = b, ne[idx] = h[a], h[a] = idx++;
}

bool dfs(int u, int c){
    color[u] = c;
    for(int i = h[u];i != -1;i = ne[i]){
        int j = e[i];
        if(!color[j]){
            if(!dfs(j, 3 - c)) return false;
        }
        else if(color[j] == c) return false;
    }
    return true;
}

void solve(){
    memset(h, -1, sizeof h);
    cin >> n >> m;
    while(m--){
        int a, b;
        cin >> a >> b;
        add(a, b), add(b, a);
    }
    bool flag = true;
    for(int i = 1;i <= n;++i){
        if(!color[i] && !dfs(i, 1)){
            flag = false;
            break;
        }
    }
    if(flag) cout << "Yes" << '\n';
    else cout << "No" << '\n';
}
```



### 匈牙利算法

二分图的最大匹配

> 二分图的匹配：给定一个二分图 G，在 G 的一个子图 M 中，M 的边集 {E} 中的任意两条边都不依附于同一个顶点，则称 M 是一个匹配。
>
> 二分图的最大匹配：所有匹配中包含边数最多的一组匹配被称为二分图的最大匹配，其边数即为最大匹配数。

```cpp
constexpr int N = 520, M = 1e5+10;

int n1, n2, m;
int h[N], e[M], ne[M], idx;
int match[N];
bool st[N];

void add(int a, int b){
    e[idx] = b, ne[idx] = h[a], h[a] = idx++;
}

bool find(int x){
    for(int i = h[x];i != -1;i = ne[i]){
        int j = e[i];
        if(!st[j]){
            st[j] = true;
            if(!match[j] || find(match[j])){
                match[j] = x; // 月老别忘了牵红线
                return true;
            }
        }
    }
    return false;
}

void solve(){
    memset(h, -1, sizeof h);
    cin >> n1 >> n2 >> m;
    while(m--){
        int a, b;
        cin >> a >> b;
        add(a, b);
    }
    int ans = 0;
    for(int i = 1;i <= n1;++i){
        memset(st, 0, sizeof st); // st是防止递归的时候找了找过的
        if(find(i)) ans++; // 能改到就加，一直都是最优解
    }
    cout << ans << '\n';
}
```
