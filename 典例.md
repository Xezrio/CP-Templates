### 分治

Secret Cow Code S

```cpp
void solve(){
    string s;
    ll n;
    cin >> s >> n;
    ll sz = s.size();
    s = " " + s;
    while(sz < n){
        ll l = sz;
        while(l < n) l *= 2;
        n = n - l / 2 - 1;
        if(!n) n = l / 2;
    }
    cout << s[n] << '\n';
}
```
